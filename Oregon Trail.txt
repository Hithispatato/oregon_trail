    answer = ""
    if money >= 1000:
        rint = random.randint(1, 5)
        if rint == 1:
            answer = "$100 was stolen."
    elif (money > 500) and (money < 1000):
        rint = random.randint(1, 7)
        if rint == 1:
            answer = "$75 was stolen."
    else:
        rint = random.randint(1, 10)
        if rint == 1:
            answer = "$50 was stolen."
    return answer
